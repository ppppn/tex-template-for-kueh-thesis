          \begin{itemize}
            \item{著者名「論文名」『雑誌名』、巻号、頁。}

      \end{itemize}
    \item{欧語}
      \begin{itemize}
        \item{書籍}

            \item{Author, {\it Title}, Publisher / Published Place, Year. }
          \end{itemize}
        \item{論文}
          \begin{itemize}
            \item{Author, “Article Title” in: Book Editors(eds.), {\it Book Title}, Year, pp. pages.}
